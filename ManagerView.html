    <!-- Firebase SDKs -->

    <script type="module">

        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";

        import { getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";

        import { getFirestore, doc, updateDoc, onSnapshot, collection, query, where } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";


        // IMPORTANT: The global variables __app_id, __firebase_config, and __initial_auth_token are provided by the environment.

        const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';

        const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : null;

        const initialAuthToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : null;


        let db;

        let auth;

        let userId;


        // --- Utility Functions ---


        /**

         * Converts currency number to a formatted string.

         * @param {number} amount - The amount to format.

         */

        const formatCurrency = (amount) => {

            const numAmount = typeof amount === 'string' ? parseFloat(amount) : amount;

            if (isNaN(numAmount)) return '$0.00';

            return new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD' }).format(numAmount);

        };


        /**

         * Displays a temporary message box instead of using alert().

         * @param {string} message - The message to display.

         * @param {string} type - 'success' or 'error'.

         */

        const showMessage = (message, type = 'success') => {

            const box = document.getElementById('message-box');

            box.textContent = message;

            box.className = 'fixed top-4 right-4 z-50 p-4 rounded-xl shadow-2xl font-medium transition-transform transform duration-300';


            if (type === 'success') {

                box.classList.add('bg-green-600', 'translate-x-0');

            } else {

                box.classList.add('bg-red-600', 'translate-x-0');

            }


            // Show for 4 seconds then hide

            setTimeout(() => {

                box.classList.remove('translate-x-0');

                box.classList.add('translate-x-full');

            }, 4000);

        };



        /**

         * Initializes Firebase and authenticates the user.

         */

        async function initializeAppAndAuth() {

            const loadingOverlay = document.getElementById('loading-overlay');

            const userIdDisplay = document.getElementById('user-id-display');


            if (!firebaseConfig) {

                console.error("Firebase config is missing.");

                loadingOverlay.remove();

                showMessage("Initialization failed: Firebase config missing.", 'error');

                return;

            }


            try {

                const app = initializeApp(firebaseConfig);

                db = getFirestore(app);

                auth = getAuth(app);


                if (initialAuthToken) {

                    await signInWithCustomToken(auth, initialAuthToken);

                } else {

                    await signInAnonymously(auth);

                }


                onAuthStateChanged(auth, (user) => {

                    if (user) {

                        userId = user.uid;

                        userIdDisplay.textContent = userId;

                        listenForRequests();

                    } else {

                        console.log("No user is signed in.");

                        userIdDisplay.textContent = 'Guest/Unauthenticated';

                    }

                    setTimeout(() => { loadingOverlay.remove(); }, 500);

                });


            } catch (error) {

                console.error("Firebase Initialization or Auth Error:", error);

                loadingOverlay.remove();

                showMessage(`Authentication failed: ${error.message}`, 'error');

            }

        }


        /**

         * Renders the list of requests in the table.

         * @param {Array<Object>} requests - The array of request documents.

         */

        function renderRequests(requests) {

            const list = document.getElementById('requests-list');

            list.innerHTML = '';


            const pendingRequests = requests.filter(req => req.status === 'Pending');


            if (pendingRequests.length === 0) {

                list.innerHTML = `

                    <tr>

                        <td colspan="6" class="p-12 text-center text-gray-500 text-xl italic font-light">

                            <span class="flex items-center justify-center space-x-3 text-green-600">

                                ${lucide.createIcons()['check-circle'].toSvg({ class: 'w-7 h-7' })}

                                <span>Review queue is empty. Grab a coffee!</span>

                            </span>

                        </td>

                    </tr>

                `;

                return;

            }


            pendingRequests.forEach(request => {

                const row = document.createElement('tr');

                row.className = 'border-b border-gray-100 transition-colors duration-200 hover:bg-indigo-50/50';


                // Status and Category styling setup (always Pending in this view)

                const statusClass = 'bg-yellow-100 text-yellow-800';

                const statusText = 'Pending';

                

                let categoryColor = 'bg-gray-200 text-gray-800';

                if (request.category === 'Travel') categoryColor = 'bg-indigo-100 text-indigo-800';

                else if (request.category === 'Meals') categoryColor = 'bg-amber-100 text-amber-800';

                else if (request.category === 'Software') categoryColor = 'bg-purple-100 text-purple-800';

                else if (request.category === 'Office Supplies') categoryColor = 'bg-cyan-100 text-cyan-800';



                row.innerHTML = `

                    <td class="px-6 py-4 whitespace-nowrap text-sm font-semibold text-gray-900">

                        ${request.description.substring(0, 40) + (request.description.length > 40 ? '...' : '') || 'Expense Request'}

                        <div class="text-xs text-gray-400 font-normal mt-1">

                            ${new Date(request.submittedAt).toLocaleDateString('en-US', { year: 'numeric', month: 'short', day: 'numeric' })}

                        </div>

                    </td>

                    <td class="px-6 py-4 whitespace-nowrap text-sm text-gray-600">

                        <span title="Employee ID: ${request.employeeId || 'Unknown'}" class="cursor-help font-mono text-xs p-1 bg-gray-50 rounded">${request.employeeId ? request.employeeId.substring(0, 8) + '...' : 'Unknown'}</span>

                    </td>

                    <td class="px-6 py-4 whitespace-nowrap text-sm">

                        <span class="inline-flex items-center px-3 py-1 rounded-full text-xs font-semibold ${categoryColor}">

                            ${request.category || 'Other'}

                        </span>

                    </td>

                    <td class="px-6 py-4 whitespace-nowrap text-center text-sm font-semibold">

                        <span class="inline-flex items-center px-3 py-1 rounded-full text-xs font-bold ${statusClass}">

                            ${statusText}

                        </span>

                    </td>

                    <td class="px-6 py-4 whitespace-nowrap amount-cell text-gray-900 font-extrabold">

                        ${formatCurrency(request.amount)}

                        <div class="text-xs text-gray-500 font-normal mt-1">

                            Vendor: ${request.vendor || 'N/A'}

                        </div>

                    </td>

                    <td class="px-6 py-4 whitespace-nowrap text-center text-sm font-medium">

                        <div data-id="actions-${request.id}" class="action-buttons flex justify-center space-x-3">

                            <button data-id="${request.id}" data-action="approve"

                                    class="action-button inline-flex items-center px-4 py-2 rounded-lg shadow-sm text-sm border-green-500 text-green-600 bg-white hover:bg-green-500 hover:text-white"

                                    title="Approve Request">

                                Approve

                            </button>

                            <button data-id="${request.id}" data-action="reject"

                                    class="action-button inline-flex items-center px-4 py-2 rounded-lg shadow-sm text-sm border-red-500 text-red-600 bg-white hover:bg-red-500 hover:text-white"

                                    title="Reject Request">

                                Reject

                            </button>

                        </div>

                    </td>

                `;

                list.appendChild(row);

            });


            // Re-attach listeners to the new buttons

            list.querySelectorAll('button[data-action]').forEach(button => {

                button.addEventListener('click', handleApprovalAction);

            });

        }


        /**

         * Handles the click event for Approve/Reject buttons and updates Firestore.

         */

        async function handleApprovalAction(event) {

            const button = event.currentTarget;

            const requestId = button.dataset.id;

            const action = button.dataset.action;

            const newStatus = action === 'approve' ? 'Approved' : 'Rejected';


            const actionContainer = document.querySelector(`[data-id="actions-${requestId}"]`);


            if (actionContainer) {

                actionContainer.innerHTML = `<span class="inline-flex items-center px-3 py-1 rounded-full text-sm font-semibold bg-indigo-100 text-indigo-800 shadow">Processing...</span>`;

            }


            try {

                if (!userId) {

                    showMessage("Authentication error. Please refresh.", 'error');

                    return;

                }


                const requestRef = doc(db, `artifacts/${appId}/public/data/approval_requests`, requestId);


                await updateDoc(requestRef, {

                    status: newStatus,

                    managerId: userId,

                    approvedAt: new Date().toISOString()

                });


                showMessage(`Request ${requestId.substring(0, 4)}... successfully ${newStatus}!`, 'success');


            } catch (error) {

                console.error(`Error updating request ${requestId}:`, error);

                if (actionContainer) {

                     actionContainer.innerHTML = `<span class="text-red-500 font-semibold">Error!</span>`;

                }

                showMessage(`Failed to ${action} request. Try again.`, 'error');

            }

        }


        /**

         * Sets up the real-time listener for pending approval requests.

         */

        function listenForRequests() {

            if (!db) return;


            const requestsColRef = collection(db, `artifacts/${appId}/public/data/approval_requests`);


            // Query for all requests where status is 'Pending'.

            const q = query(requestsColRef, where("status", "==", "Pending"));


            onSnapshot(q, (snapshot) => {

                const requests = [];

                snapshot.forEach((doc) => {

                    requests.push({ id: doc.id, ...doc.data() });

                });

                renderRequests(requests);

            }, (error) => {

                console.error("Error listening to requests:", error);

                showMessage("Error loading real-time data.", 'error');

            });

        }


        // Initialize the application on load

        window.onload = initializeAppAndAuth;


    </script>
